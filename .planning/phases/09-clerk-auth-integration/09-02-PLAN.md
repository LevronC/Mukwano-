**Step 5: Rewrite `packages/api/src/hooks/auth-guard.ts`**
Complete replacement with Clerk verifyToken + auto-provisioning:

```typescript
import type { FastifyRequest, FastifyReply } from 'fastify'

export async function authGuard(request: FastifyRequest, reply: FastifyReply): Promise<void> {
  const authHeader = request.headers.authorization
  if (!authHeader?.startsWith('Bearer ')) {
    return reply.code(401).send({
      error: { code: 'UNAUTHORIZED', message: 'Authentication required', field: null, status: 401 }
    })
  }

  const token = authHeader.slice(7)

  try {
    const payload = await request.server.clerk.verifyToken(token, {
      secretKey: request.server.config.CLERK_SECRET_KEY
    })

    // payload.sub is the Clerk user ID
    let user = await request.server.prisma.user.findUnique({
      where: { clerkId: payload.sub }
    })

    if (!user) {
      // Auto-provision: fetch Clerk user details
      const clerkUser = await request.server.clerk.users.getUser(payload.sub)
      const primaryEmail = clerkUser.emailAddresses.find(
        (e: { id: string; emailAddress: string; verification?: { status: string } | null }) =>
          e.id === clerkUser.primaryEmailAddressId && e.verification?.status === 'verified'
      )?.emailAddress

      if (!primaryEmail) {
        return reply.code(401).send({
          error: { code: 'UNVERIFIED_EMAIL', message: 'Your primary email address must be verified before you can use Mukwano', field: null, status: 401 }
        })
      }

      // displayName fallback: firstName + lastName, or email prefix
      const displayName = `${clerkUser.firstName ?? ''} ${clerkUser.lastName ?? ''}`.trim() || primaryEmail.split('@')[0]

      // Use upsert to handle race condition (two simultaneous first requests)
      user = await request.server.prisma.user.upsert({
        where: { clerkId: payload.sub },
        create: {
          clerkId: payload.sub,
          email: primaryEmail,
          displayName,
          passwordHash: null
        },
        update: {} // no-op if already exists
      })
    }

    request.user = { id: user.id, email: user.email, isGlobalAdmin: user.isGlobalAdmin }
  } catch {
    return reply.code(401).send({
      error: { code: 'UNAUTHORIZED', message: 'Authentication required', field: null, status: 401 }
    })
  }
}
```
